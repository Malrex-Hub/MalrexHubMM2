-- safe parent so we don't get banned for touching coregui the wrong way
local function safeParent(g)
    pcall(function()
        if gethui then g.Parent = gethui(); return end
        g.Parent = CoreGui
    end)
    if not g.Parent then g.Parent = LP:WaitForChild("PlayerGui") end
end

-- tween shortcut because typing TweenService:Create every time is cancer
local function tw(obj, props, t, style)
    TweenService:Create(obj, TweenInfo.new(t or 0.2, style or Enum.EasingStyle.Quad, Enum.EasingDirection.Out), props):Play()
end

-- teleport with hold so you don't fall through the fucking map
local tpHoldConn = nil
local function tpTo(cf, holdTime)
    local root = getRoot()
    if not root then return end
    -- cf can be a CFrame or Vector3, handle both
    if typeof(cf) == "Vector3" then cf = CFrame.new(cf) end
    root.CFrame = cf
    if holdTime and holdTime > 0 then
        if tpHoldConn then tpHoldConn:Disconnect() end
        local elapsed = 0
        tpHoldConn = RunService.Heartbeat:Connect(function(dt)
            elapsed = elapsed + dt
            local r = getRoot()
            if r then r.CFrame = cf end
            if elapsed >= holdTime then
                tpHoldConn:Disconnect()
                tpHoldConn = nil
            end
        end)
    end
end

-- get all alive enemies
local function getMobs()
    local enemies = Workspace:FindFirstChild("Enemies")
    local list = {}
    if not enemies then return list end
    for _, v in ipairs(enemies:GetChildren()) do
        local h = v:FindFirstChildOfClass("Humanoid")
        local r = v:FindFirstChild("HumanoidRootPart")
        if h and r and h.Health > 0 then
            table.insert(list, v)
        end
    end
    return list
end

-- get nearest mob matching a name filter
local function getNearestMob(mobName)
    local root = getRoot()
    if not root then return nil end
    local best, bestDist = nil, math.huge
    for _, mob in ipairs(getMobs()) do
        local match = mobName == "" or string.find(mob.Name:lower(), mobName:lower(), 1, true)
        if match then
            local hrp = mob:FindFirstChild("HumanoidRootPart")
            if hrp then
                local d = (hrp.Position - root.Position).Magnitude
                if d < bestDist then bestDist = d; best = mob end
            end
        end
    end
    return best
end

-- get all mobs within a radius
local function getMobsInRadius(radius)
    local root = getRoot()
    if not root then return {} end
    local result = {}
    for _, mob in ipairs(getMobs()) do
        local hrp = mob:FindFirstChild("HumanoidRootPart")
        if hrp and (hrp.Position - root.Position).Magnitude <= radius then
            table.insert(result, mob)
        end
    end
    return result
end

-- press a key properly (not that busted string.byte bullshit)
local function pressKey(key)
    pcall(function()
        local kc = Enum.KeyCode[key:upper()]
        if kc then
            game:GetService("VirtualInputManager"):SendKeyEvent(true, kc.Value, false, nil)
            task.wait(0.04)
            game:GetService("VirtualInputManager"):SendKeyEvent(false, kc.Value, false, nil)
        end
    end)
end

-- fire the CommF remote with pcall so one failure doesn't kill everything
local function fireRemote(...)
    local args = {...}
    pcall(function()
        ReplicatedStorage.Remotes.CommF_:InvokeServer(unpack(args))
    end)
end

-- attack a specific mob target
local function attackMob(mob)
    if not mob or not mob.Parent then return end
    local hrp = mob:FindFirstChild("HumanoidRootPart")
    if not hrp then return end
    local root = getRoot()
    if not root then return end

    local target = hrp.Position
    if state.aimbot then
        target = target + (hrp.Velocity * state.prediction)
    end
    root.CFrame = CFrame.new(target + Vector3.new(0, 5, 0), target)

    -- m1 click
    pcall(function()
        game:GetService("VirtualInputManager"):SendMouseButtonEvent(0, 0, 0, true, game, 1)
        task.wait(0.04)
        game:GetService("VirtualInputManager"):SendMouseButtonEvent(0, 0, 0, false, game, 1)
    end)
    fireRemote("attackEnemy", mob)

    -- skill keys based on mode
    if state.skills then
        local keys = {}
        if state.spammode == "all" then keys = {"Z","X","C","V","F"}
        elseif state.spammode == "fruitonly" then keys = {"Z","X","C"}
        elseif state.spammode == "swordonly" then keys = {"Z","X"}
        elseif state.spammode == "gunonly"   then keys = {"Z"}
        end
        for _, k in ipairs(keys) do
            pressKey(k)
            task.wait(0.04)
        end
    end
end

local farmThread  = nil
local lastQuestMob = ""

local function startFarmLoop()
    if farmThread then task.cancel(farmThread) end
    farmThread = task.spawn(function()
        while state.farm do
            local ok, err = pcall(function()
                local hum  = getHum()
                local root = getRoot()
                if not hum or not root then task.wait(1); return end
                if hum.Health <= 0 then task.wait(2); return end

                -- eat if low
                if state.autoeat and (hum.Health / hum.MaxHealth * 100) < state.hp then
                    fireRemote("eatFruit")
                    task.wait(0.5)
                end

                -- find mob target from database
                local target = nil
                for _, m in ipairs(getCurrentMobs()) do
                    if state.mob == "" or string.find(m.name:lower(), state.mob:lower(), 1, true) then
                        target = m
                        break
                    end
                end

                -- grab quest if we need one
                if state.quest and target and target.quest and lastQuestMob ~= target.name then
                    tpTo(target.qp + Vector3.new(0, 5, 0), 0.8)
                    task.wait(0.5)
                    fireRemote("startQuest", target.quest, target.lvl)
                    lastQuestMob = target.name
                    task.wait(0.4)
                end

                -- find the actual mob in world
                local mob = getNearestMob(state.mob)
                if mob then
                    local hits = state.fastattack and 18 or 12
                    for i = 1, hits do
                        if not state.farm then break end
                        local mh = mob:FindFirstChildOfClass("Humanoid")
                        if not mh or mh.Health <= 0 then break end
                        attackMob(mob)
                        task.wait(state.fastattack and 0.06 or state.skilldelay)
                    end

                    -- check if quest is done
                    local mh2 = mob:FindFirstChildOfClass("Humanoid")
                    if mh2 and mh2.Health <= 0 then
                        state.totalkills = state.totalkills + 1
                        pcall(function()
                            local qg = LP.PlayerGui:FindFirstChild("Main")
                            if qg then
                                local done = qg:FindFirstChild("QuestComplete", true)
                                if done and done.Visible then
                                    lastQuestMob = ""
                                end
                            end
                        end)
                    end
                else
                    -- no mob found nearby, go to spawn
                    if target and target.mp then
                        tpTo(target.mp + Vector3.new(0, 8, 0), 0.5)
                        task.wait(0.5)
                    else
                        task.wait(0.3)
                    end
                end
            end)
            if not ok then
                -- something fucked up, wait a sec and try again
                task.wait(1)
            else
                task.wait(0.08)
            end
        end
    end)
end

                            if not state.bosshunt then break end

                            end

                    local bList = bossData[sea] or bossData[1]
                    for _, b in ipairs(bList) do
                        if b.name:lower():find(state.selectedboss:lower()) then
                            tpTo(b.cf + Vector3.new(0, 5, 0), 1)
                            break
                        end
                    end
                    task.wait(3)
                end
            end)
            task.wait(0.3)
        end
    end)

-- ===========================================================================
-- ESP SYSTEM (see players through walls like a real cheater)
-- ===========================================================================
local espObjects = {}
local espConnections = {}

local function clearESP()
    for _, v in pairs(espObjects) do pcall(function() v:Destroy() end) end
    for _, v in pairs(espConnections) do pcall(function() v:Disconnect() end) end
    espObjects = {}
    espConnections = {}
end

local function addESP(player)
    if player == LP then return end
    local function buildESP()
        local char = player.Character
        if not char then return end
        local rootPart = char:FindFirstChild("HumanoidRootPart")
        if not rootPart then return end

        -- clean up old one if it exists
        if espObjects[player.Name] then
            pcall(function() espObjects[player.Name]:Destroy() end)
        end

        local bb = Instance.new("BillboardGui")
        bb.Size = UDim2.new(0, 150, 0, 60)
        bb.StudsOffset = Vector3.new(0, 4, 0)
        bb.AlwaysOnTop = true
        bb.Adornee = rootPart
        safeParent(bb)

        local bg = Instance.new("Frame")
        bg.Size = UDim2.new(1, 0, 1, 0)
        bg.BackgroundColor3 = Color3.fromRGB(8, 8, 16)
        bg.BackgroundTransparency = 0.4
        bg.BorderSizePixel = 0
        bg.Parent = bb
        local bgCr = Instance.new("UICorner")
        bgCr.CornerRadius = UDim.new(0, 8)
        bgCr.Parent = bg

        local espColor = state.teamcolor and (player.TeamColor and player.TeamColor.Color or state.espcolor) or state.espcolor
        local bgSt = Instance.new("UIStroke")
        bgSt.Color = espColor
        bgSt.Thickness = 1.5
        bgSt.Parent = bg

        -- name
        local nameL = Instance.new("TextLabel")
        nameL.Size = UDim2.new(1, -6, 0, 20)
        nameL.Position = UDim2.new(0, 3, 0, 3)
        nameL.BackgroundTransparency = 1
        nameL.Text = player.Name
        nameL.TextColor3 = espColor
        nameL.TextStrokeTransparency = 0.5
        nameL.Font = Enum.Font.GothamBold
        nameL.TextSize = 13
        nameL.Parent = bg

        -- distance
        local distL = Instance.new("TextLabel")
        distL.Size = UDim2.new(1, -6, 0, 14)
        distL.Position = UDim2.new(0, 3, 0, 24)
        distL.BackgroundTransparency = 1
        distL.TextColor3 = Color3.fromRGB(200, 200, 210)
        distL.Font = Enum.Font.Gotham
        distL.TextSize = 10
        distL.Parent = bg

        -- health bar track
        local hpTrack = Instance.new("Frame")
        hpTrack.Size = UDim2.new(1, -10, 0, 5)
        hpTrack.Position = UDim2.new(0, 5, 1, -9)
        hpTrack.BackgroundColor3 = Color3.fromRGB(50, 10, 10)
        hpTrack.BorderSizePixel = 0
        hpTrack.Parent = bg
        local htCr = Instance.new("UICorner")
        htCr.CornerRadius = UDim.new(1, 0)
        htCr.Parent = hpTrack

        local hpFill = Instance.new("Frame")
        hpFill.Size = UDim2.new(1, 0, 1, 0)
        hpFill.BackgroundColor3 = Color3.fromRGB(40, 200, 80)
        hpFill.BorderSizePixel = 0
        hpFill.Parent = hpTrack
        local hfCr = Instance.new("UICorner")
        hfCr.CornerRadius = UDim.new(1, 0)
        hfCr.Parent = hpFill

        espObjects[player.Name] = bb

        local conn
        conn = RunService.Heartbeat:Connect(function()
            if not state.esp or not player.Character or not bb.Parent then
                pcall(function() bb:Destroy() end)
                espObjects[player.Name] = nil
                conn:Disconnect()
                return
            end
            local myRoot = getRoot()
            if myRoot and player.Character:FindFirstChild("HumanoidRootPart") then
                local dist = (player.Character.HumanoidRootPart.Position - myRoot.Position).Magnitude
                distL.Text = math.floor(dist) .. "m away"
            end
            local hum = player.Character:FindFirstChildOfClass("Humanoid")
            if hum and hum.MaxHealth > 0 then
                local pct = hum.Health / hum.MaxHealth
                hpFill.Size = UDim2.new(math.clamp(pct, 0, 1), 0, 1, 0)
                hpFill.BackgroundColor3 = pct > 0.6 and Color3.fromRGB(40,200,80)
                    or pct > 0.3 and Color3.fromRGB(255,200,30)
                    or Color3.fromRGB(220,50,50)
            end
        end)
        espConnections[player.Name] = conn
    end

    if player.Character then buildESP() end
    player.CharacterAdded:Connect(function() task.wait(1); buildESP() end)
end

local function refreshESP()
    clearESP()
    if not state.esp then return end
    for _, p in ipairs(Players:GetPlayers()) do addESP(p) end
end

Players.PlayerAdded:Connect(function(p) if state.esp then addESP(p) end end)
Players.PlayerRemoving:Connect(function(p)
    if espObjects[p.Name] then espObjects[p.Name]:Destroy(); espObjects[p.Name] = nil end
    if espConnections[p.Name] then espConnections[p.Name]:Disconnect(); espConnections[p.Name] = nil end
end)

                ln.CFrame = CFrame.new(sPos, tPos) * CFrame.new(0, 0, -dist / 2)
            end
        end
        for name, ln in pairs(tracerLines) do
            if not Players:FindFirstChild(name) then
                ln:Destroy()
                tracerLines[name] = nil
            end
        end

local function createTab(name)
    local page = Instance.new("ScrollingFrame")
    page.Size = UDim2.new(1,-8,1,-8)
    page.Position = UDim2.new(0,4,0,4)
    page.BackgroundTransparency = 1
    page.BorderSizePixel = 0
    page.ScrollBarThickness = 4
    page.ScrollBarImageColor3 = colors.accent
    page.CanvasSize = UDim2.new(0,0,0,0)
    page.AutomaticCanvasSize = Enum.AutomaticSize.Y
    page.Visible = false
    page.Parent = contentArea

    local pageLayout = Instance.new("UIListLayout")
    pageLayout.Padding = UDim.new(0,5)
    pageLayout.Parent = page

    local pagePad = Instance.new("UIPadding")
    pagePad.PaddingTop    = UDim.new(0,5)
    pagePad.PaddingLeft   = UDim.new(0,5)
    pagePad.PaddingRight  = UDim.new(0,5)
    pagePad.PaddingBottom = UDim.new(0,5)
    pagePad.Parent = page

    -- sidebar tab button
    local tabBtn = Instance.new("TextButton")
    tabBtn.Size = UDim2.new(1,0,0,38)
    tabBtn.BackgroundColor3 = colors.bg3
    tabBtn.BackgroundTransparency = 1
    tabBtn.BorderSizePixel = 0
    tabBtn.Text = ""
    tabBtn.AutoButtonColor = false
    tabBtn.Parent = tabScroll
    mkCorner(tabBtn, 8)

    local indicator = mkFrame(tabBtn, UDim2.new(0,3,0,20), UDim2.new(0,0,0.5,-10), colors.accent)
    mkCorner(indicator, 2)
    indicator.Visible = false

    local iconBox = mkFrame(tabBtn, UDim2.new(0,22,0,22), UDim2.new(0,8,0.5,-11), colors.bg)
    mkCorner(iconBox, 6)
    local iconL = mkLabel(iconBox, TAB_ICONS[name] or name:sub(1,1), UDim2.new(1,0,1,0), nil, colors.textD, Enum.Font.GothamBold, Enum.TextXAlignment.Center)
    iconL.TextSize = 10

    local tabLabelL = mkLabel(tabBtn, name, UDim2.new(1,-42,1,0), UDim2.new(0,36,0,0), colors.textD, Enum.Font.GothamMedium)
    tabLabelL.TextSize = 12

    local function activateTab()
        if activeTabName == name then return end
        -- deactivate old
        if activeTabName and tabRegistry[activeTabName] then
            local old = tabRegistry[activeTabName]
            old.page.Visible = false
            old.ind.Visible  = false
            tw(old.btn, {BackgroundTransparency=1}, 0.12)
            tw(old.label, {TextColor3=colors.textD}, 0.12)
            tw(old.icon, {TextColor3=colors.textD}, 0.12)
            tw(old.iconBox, {BackgroundColor3=colors.bg}, 0.12)
        end
        activeTabName = name
        page.Visible  = true
        indicator.Visible = true
        tw(tabBtn, {BackgroundTransparency=0, BackgroundColor3=Color3.fromRGB(20,10,44)}, 0.12)
        tw(tabLabelL, {TextColor3=colors.text}, 0.12)
        tw(iconL, {TextColor3=colors.accentL}, 0.12)
        tw(iconBox, {BackgroundColor3=Color3.fromRGB(40,20,80)}, 0.12)
    end

    tabBtn.MouseButton1Click:Connect(activateTab)
    tabBtn.MouseEnter:Connect(function()
        if activeTabName ~= name then tw(tabBtn, {BackgroundTransparency=0.82}, 0.1) end
    end)
    tabBtn.MouseLeave:Connect(function()
        if activeTabName ~= name then tw(tabBtn, {BackgroundTransparency=1}, 0.1) end
    end)

    tabRegistry[name] = {page=page, btn=tabBtn, label=tabLabelL, ind=indicator, iconBox=iconBox, icon=iconL, activate=activateTab}
    if not activeTabName then activateTab() end

        local toggled = default or false

        bc.ZIndex = 6
        bc.MouseEnter:Connect(function() tw(w, {BackgroundColor3=colors.hover}, 0.1) end)
        bc.MouseLeave:Connect(function() tw(w, {BackgroundColor3=colors.bg3},  0.1) end)
        bc.MouseButton1Click:Connect(function() set(not toggled) end)

        return {SetValue=function(_, v) set(v) end, GetValue=function() return toggled end}
    end

    function Tab:Dropdown(title, options, cb)
        local selected = options[1] or ""
        local isOpen   = false
        local w = mkFrame(page, UDim2.new(1,0,0,50), nil, colors.bg3)
        w.LayoutOrder = nextOrd()
        w.ClipsDescendants = false
        mkCorner(w, 8)
        mkStroke(w, colors.border)

        local acBar = mkFrame(w, UDim2.new(0,3,0,32), UDim2.new(0,0,0,9), colors.accent)
        mkCorner(acBar, 2)
        acBar.ZIndex = 2

        local tl = mkLabel(w, title, UDim2.new(0.6,0,0,15), UDim2.new(0,12,0,5), colors.textD, Enum.Font.Gotham)
        tl.TextSize = 11
        tl.ZIndex = 2

        local selBtn = Instance.new("TextButton")

        selBtn.AutoButtonColor = false
        selBtn.ZIndex = 3
        selBtn.Parent = w
        mkCorner(selBtn, 6)
        mkStroke(selBtn, colors.border)

        local selL = mkLabel(selBtn, selected, UDim2.new(1,-26,1,0), UDim2.new(0,8,0,0), colors.text, Enum.Font.Gotham)
        selL.TextSize = 11

            isOpen = false
            tw(arrow, {Rotation=0}, 0.12)
            tw(panel, {Size=UDim2.new(0, panel.AbsoluteSize.X, 0, 0)}, 0.14)
            task.wait(0.17)
            panel.Visible = false
        end

        local function buildItems()
            for _, ch in ipairs(panel:GetChildren()) do
                if ch:IsA("TextButton") then ch:Destroy() end
            end
            for _, opt in ipairs(options) do
                local item = Instance.new("TextButton")
                item.Size = UDim2.new(1,0,0,28)
                item.BackgroundColor3 = opt==selected and Color3.fromRGB(30,12,65) or colors.bg2

    function Tab:Slider(title, minVal, maxVal, defaultVal, cb)
        local val = defaultVal or minVal
        local w = mkFrame(page, UDim2.new(1,0,0,58), nil, colors.bg3)
        w.LayoutOrder = nextOrd()
        mkCorner(w, 8)
        mkStroke(w, colors.border)

        mkFrame(w, UDim2.new(0,3,0,40), UDim2.new(0,0,0,9), colors.accent)

        local tl = mkLabel(w, title, UDim2.new(1,-70,0,20), UDim2.new(0,12,0,7), colors.text, Enum.Font.GothamMedium)
        tl.TextSize = 12

        local vl = mkLabel(w, tostring(val), UDim2.new(0,56,0,20), UDim2.new(1,-60,0,7), colors.accentL, Enum.Font.GothamBold, Enum.TextXAlignment.Right)
        vl.TextSize = 12

        local track = mkFrame(w, UDim2.new(1,-24,0,6), UDim2.new(0,12,0,40), colors.off)
        mkCorner(track, 4)

        local fill = mkFrame(track, UDim2.new((val-minVal)/(maxVal-minVal),0,1,0), nil, colors.accent)
        mkCorner(fill, 4)

        local knob = mkFrame(fill, UDim2.new(0,14,0,14), UDim2.new(1,-7,0.5,-7), Color3.new(1,1,1))
        mkCorner(knob, 7)

        local sliding = false

        local function doUpdate(x)
            local pct = math.clamp((x - track.AbsolutePosition.X) / track.AbsoluteSize.X, 0, 1)
            val = math.round(minVal + (maxVal-minVal)*pct)
            vl.Text = tostring(val)
            fill.Size = UDim2.new(pct, 0, 1, 0)
            pcall(cb, val)
        end

        track.InputBegan:Connect(function(i)
            if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
                sliding = true; doUpdate(i.Position.X)
            end
        end)
        UserInputService.InputChanged:Connect(function(i)
            if sliding and (i.UserInputType == Enum.UserInputType.MouseMovement or i.UserInputType == Enum.UserInputType.Touch) then
                doUpdate(i.Position.X)
            end
        end)
        UserInputService.InputEnded:Connect(function(i)
            if i.UserInputType == Enum.UserInputType.MouseButton1 or i.UserInputType == Enum.UserInputType.Touch then
                sliding = false
            end
        end)
    end

    function Tab:Input(title, placeholder, default, cb)
        local val = default or ""
        local w = mkFrame(page, UDim2.new(1,0,0,50), nil, colors.bg3)
        w.LayoutOrder = nextOrd()
        mkCorner(w, 8)
        mkStroke(w, colors.border)

        mkFrame(w, UDim2.new(0,3,0,32), UDim2.new(0,0,0,9), colors.accent)

        local tl = mkLabel(w, title, UDim2.new(1,-14,0,16), UDim2.new(0,12,0,5), colors.textD, Enum.Font.GothamMedium)
        tl.TextSize = 11

        local box = Instance.new("TextBox")
        box.Size = UDim2.new(1,-24,0,22)
        box.Position = UDim2.new(0,12,0,25)
        box.BackgroundColor3 = colors.bg
        box.BorderSizePixel = 0
        box.Text = val
        box.PlaceholderText = placeholder or ""
        box.PlaceholderColor3 = colors.textD
        box.TextColor3 = colors.text
        box.Font = Enum.Font.Gotham
        box.TextSize = 11
        box.ZIndex = 3
        box.Parent = w
        mkCorner(box, 6)
        mkStroke(box, colors.border)

        box.FocusLost:Connect(function(enter)
            if enter then val = box.Text; pcall(cb, val) end
        end)

        return {SetValue=function(_, v) val=v; box.Text=v; pcall(cb,v) end}
    end

    return Tab
end

local minimized = false
UserInputService.InputBegan:Connect(function(i, processed)
    if processed then return end

    if i.KeyCode == Enum.KeyCode.End then
        minimized = not minimized
        contentArea.Visible = not minimized
        sidebar.Visible     = not minimized
        tw(mainWin, {Size = minimized and UDim2.fromOffset(200,52) or winSize}, 0.25, Enum.EasingStyle.Quint)
        if minimized then notify("Hub minimized (END to restore)", 2, "info") end
    end

    if i.KeyCode == Enum.KeyCode.Home then
        state.farm = not state.farm
        if state.farm then
            startFarmLoop()
            notify("Farm ON (Home)", 2, "good")
        else
            notify("Farm OFF (Home)", 2, "bad")
        end
    end

    if i.KeyCode == Enum.KeyCode.Insert then
        state.esp = not state.esp
        refreshESP()
        notify("ESP " .. (state.esp and "ON" or "OFF") .. " (Insert)", 2)
    end

    if i.KeyCode == Enum.KeyCode.Delete then
        state.noclip = not state.noclip
        startNoclip()
        notify("Noclip " .. (state.noclip and "ON" or "OFF") .. " (Delete)", 2)
    end

    if i.KeyCode == Enum.KeyCode.PageUp then
        state.walkspeed = math.min(state.walkspeed + 10, 500)
        setSpeed()
        notify("Speed: " .. state.walkspeed, 1.5)
    end

    if i.KeyCode == Enum.KeyCode.PageDown then
        state.walkspeed = math.max(state.walkspeed - 10, 16)
        setSpeed()
        notify("Speed: " .. state.walkspeed, 1.5)
    end
end)
